Given an array of `integers arr`, and `three integers` `a, b and c`. You need to find the number of `good triplets`.

A triplet (`arr[i], arr[j], arr[k]`) is good if the following conditions are true:

`0 <= i < j < k < arr`.length
`|arr[i] - arr[j]| <= a`
`|arr[j] - arr[k]| <= b`
`|arr[i] - arr[k]| <= c`
Where `|x| denotes` the absolute `value of x`.

Return the number of `good triplets`.

class Solution {
    public int countGoodTriplets(int[] arr, int a, int b, int c){
      int i,j,k,count=0;
      for(i=0;i<(arr.length)-2;i++){
        for(j=i+1;j<(arr.length)-1;j++){
          if(Math.abs(arr[i]-arr[j]) <=a){
            for(k=j+1;k<arr.length;k++){
             if(Math.abs(arr[j]-arr[k]) <=b && Math.abs(arr[i]-arr[k]) <=c)
                  count++;
            }
          }
        }
      }
      return count;
    }
}

            for (int j = i + 1; j < length; ++j) {
                if (abs(arr[i] - arr[j]) <= a) {

class Solution:
    def countGoodTriplets(self, arr: List[int], a: int, b: int, c: int) -> int:
        good_triplets = 0
        length = len(arr)

        for i in range(length):  # i is the first index
            for j in range(i + 1, length):  # j must be after i
                if abs(arr[i] - arr[j]) <= a:  # first condition check
                    for k in range(j + 1, length):  # k must be after j
                        # Check all three conditions
                        if abs(arr[j] - arr[k]) <= b and abs(arr[i] - arr[k]) <= c:
                            good_triplets += 1
                            # ğŸ¥‹ Rock Lee: "Another success from persistence!" ğŸ¥³
        
        return good_triplets
